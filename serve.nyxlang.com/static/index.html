Composable

Middleware that composes

Logging, CORS, auth — stack them in order.

// Built-in middleware
app_use(app, mw_logging())    // logs method, path, status, latency
app_use(app, mw_cors())       // Access-Control-Allow-Origin: *

// Configure CORS
cors_configure("https://myapp.com", "GET,POST", "Content-Type")

// Custom middleware
fn auth_check(req: Request, resp: Response) -> int {
    if req.headers.get("Authorization") == "" {
        response_status(resp, 401)
        return 0  // stop chain
    }
    return 1      // continue
}
APIs

JSON APIs in seconds

Parse request bodies, return JSON with proper headers.

fn handle_users(req: Request, resp: Response) {
    // Parse JSON body
    let body: Map = req_json(req)

    // Return JSON response
    let data: Map = Map.new()
    data.insert("name", "Alice")
    data.insert("role", "admin")
    response_json_map(resp, data)
    // → {"name":"Alice","role":"admin"}
    // Content-Type: application/json set automatically
}
State

Sessions out of the box

Cookie-based sessions backed by NyxKV via RESP over TCP.

import "std/session"

session_configure("127.0.0.1", 6380, 3600)

fn handle_login(req: Request, resp: Response) {
    let sid: String = session_start(req, resp)
    session_set(req, "user", "alice")
    response_json(resp, "{\"ok\":true}")
}

fn handle_profile(req: Request, resp: Response) {
    let user: String = session_get(req, "user")
    response_json(resp, "{\"user\":\"" + user + "\"}")
}
