Middleware que se compone

Logging, CORS, auth — apilalos en orden.

// Middleware integrado
app_use(app, mw_logging())    // registra method, path, status, latencia
app_use(app, mw_cors())       // Access-Control-Allow-Origin: *

// Middleware personalizado
fn auth_check(req: Request, resp: Response) -> int {
    if req.headers.get("Authorization") == "" {
        response_status(resp, 401)
        return 0  // detener cadena
    }
    return 1      // continuar
}

Sesiones integradas

Sesiones basadas en cookies respaldadas por NyxKV via RESP sobre TCP.

import "std/session"

session_configure("127.0.0.1", 6380, 3600)

fn handle_login(req: Request, resp: Response) {
    let sid: String = session_start(req, resp)
    session_set(req, "usuario", "alicia")
    response_json(resp, "{\"ok\":true}")
}
